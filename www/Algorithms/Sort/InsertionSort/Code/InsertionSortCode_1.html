            func InsertionSorted(for arr: [Int], lastIndex: Int) -> [Int] {
                var array = arr
                
                var key: Int
                var index: Int
                
                for unsortedIndex in 1...lastIndex {
                    key = array[unsortedIndex]
                    index = unsortedIndex - 1
                    
                    while index >= 0 && array[index] > key {
                        array[index+1] = array[index]
                        index -= 1
                    }
                    array[index+1] = key
                }
                return array
            }
