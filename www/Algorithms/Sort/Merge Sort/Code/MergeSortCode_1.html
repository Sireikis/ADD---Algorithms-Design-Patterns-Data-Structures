            func mergeSort(_ array: [Int]) -> [Int] {
                if array.count <= 1 {
                    return array
                }
                
                let middleIndex = array.count / 2
                
                let leftArray = mergeSort(Array(array[0..<middleIndex]))
                let rightArray = mergeSort(Array(array[middleIndex..<array.count]))
                
                return merge(leftArray, rightArray)
            }
            
            func merge(_ left: [Int], _ right: [Int]) -> [Int] {
                var leftIndex = 0
                var rightIndex = 0
                
                var orderedArray: [Int] = []
                
                while leftIndex < left.count && rightIndex < right.count {
                    let leftElement = left[leftIndex]
                    let rightElement = right[rightIndex]
                    
                    if leftElement <= rightElement {
                        orderedArray.append(leftElement)
                        leftIndex += 1
                    } else {
                        orderedArray.append(rightElement)
                        rightIndex += 1
                    }
                }
                
                while leftIndex < left.count {
                    orderedArray.append(left[leftIndex])
                    leftIndex += 1
                }
                while rightIndex < right.count {
                    orderedArray.append(right[rightIndex])
                    rightIndex += 1
                }
                return orderedArray
            }

Succinct


1: If the array is less than two elements, return it
2: Find the middle index of the array
3: Repeat 1-2 with a sub-array from index 0 up to but not including the middle index
4: The returned array is the left array
5: Repeat 1-4 with a sub-array from the middle index up to the last index
6: The returned array is the right array
7: Merge the left and right arrays

Verbose


Merge Sort Function: Expects an array
In a zero-based array,
If the array is 2 or more elements,
    Find the middle index
    Declare a left array, returned by recursively calling Merge Sort with a subarray from index 0 up to but not including the middle index
    Declare a right array, returned by recursively calling Merge Sort with a subarray from the middle index up to the last index
    
    Return the result of the Merge Function
Otherwise,
    Return the array

    
Merge Function: Expects a left and right array
In a zero-based array,
Declare leftIndex and rightIndex to be 0
Declare an empty orderedArray

While the leftIndex and rightIndex don't exceed their respective arrays,
    Declare leftElement to be the element at leftIndex
    Declare rightElement to be the element at rightIndex
    
    If leftElement is less than or equal to rightElement,
        Append leftElement to orderedArray
        Increment leftIndex
    Otherwise,
        Append rightIndex to orderedArray
        Increment rightIndex

While leftIndex is less than left array's length,
    Append the element at leftIndex to orderedArray
    Increment leftIndex

While rightIndex is less than right array's length,
    Append the element at rightIndex to orderedArray
    Increment rightIndex

Return orderedArray
