Bridge Pattern

UML


            // Interface - Abstract Implementor
            protocol Implementor {
                func someOperation()
            }

            // -------------------------

            // An Abstraction can use any implementor
            class Abstraction {
                private var implementor: Implementor
                
                init(implementor: Implementor) {
                    self.implementor = implementor
                }
                
                func useSomeOperation() {
                    implementor.someOperation()
                }
                
                func changeImplementation(to implementor: Implementor) {
                    self.implementor = implementor
                }
            }

            // Concrete Implementors
            // Any Implementor can be used by any Abstraction

            class ConcreteImplementorA: Implementor {
                func someOperation() {
                    print("Some operation A")
                }
            }

            class ConcreteImplementorB: Implementor {
                func someOperation() {
                    print("Some operation B")
                }
            }

            class ConcreteImplementorC: Implementor {
                func someOperation() {
                    print("Some operation C")
                }
            }

            // -------------------------

            var implementorA = ConcreteImplementorA()
            var implementorB = ConcreteImplementorB()
            var implementorC = ConcreteImplementorC()

            var abstraction = Abstraction(implementor: implementorA)
            abstraction.useSomeOperation() // "Some operation A"

            abstraction.changeImplementation(to: implementorB)
            abstraction.useSomeOperation() // "Some operation B"

            abstraction.changeImplementation(to: implementorC)
            abstraction.useSomeOperation() // "Some operation C"
