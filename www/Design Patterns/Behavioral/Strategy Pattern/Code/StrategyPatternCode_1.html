Strategy Pattern

UML Code


            // Could be an Abstract Class or used by the Client
            // Contains the interface for interacting with a Strategy
            class Context {
                private var strategy: Strategy
                
                public func executeStrategy() {
                    strategy.execute()
                }
                
                public func changeStrategy(to strategy: Strategy) {
                    self.strategy = strategy
                }
                
                init(strategy: Strategy) {
                    self.strategy = strategy
                }
                
                // Other Methods
            }

            // Interface for family of algorithms that vary behavior
            protocol Strategy {
                func execute()
            }

            // Concrete Class - implementation of Strategy
            // Defines one of the various strategy behaviors that are interchangeable
            class SomeBehavior: Strategy {
                public func execute() {
                    print("I have some behavior")
                }
            }

            // Concrete Class - implementation of Strategy
            // Another different strategy behavior
            class SomeOtherBehavior: Strategy {
                public func execute() {
                    print("I have some other behavior")
                }
            }

            // --------------------------

            // Context used by Client with some behavior set at runtime
            var client = Context(strategy: SomeBehavior())
            client.executeStrategy()  // "I have some behavior"

            // Context behavior dynamically changed at runtime
            client.changeStrategy(to: SomeOtherBehavior())
            client.executeStrategy()  // "I have some other behavior"
