Strategy Pattern

UML Skeleton


            // Abstract Class
            class Client {
                var behavior: IStrategy
                
                func performBehavior() {
                    behavior.strategy()
                }
                
                func changeBehavior(to behavior: IStrategy) {
                    self.behavior = behavior
                }
                
                init(behavior: IStrategy) {
                    self.behavior = behavior
                }
            }

            // Interface for family of algorithms that vary behavior
            protocol IStrategy {
                func strategy()
            }

            // Concrete Class - implementation of IStrategy
            // Defines one of the various strategy behaviors that are interchangeable
            class SomeBehavior: IStrategy {
                func strategy() {
                    print("I have some behavior")
                }
            }

            // Concrete Class - implementation of IStrategy
            // Another different strategy behavior
            class SomeOtherBehavior: IStrategy {
                func strategy() {
                    print("I have some other behavior")
                }
            }

            // --------------------------

            // Concrete Class - implementation of Client
            class ConcreteClient: Client {
                // Other Methods
            }

            // Concrete Client with some behavior set at runtime
            var client = ConcreteClient(behavior: SomeBehavior())
            client.performBehavior()  // "I have some behavior"

            // Concrete Client behavior dynamically changed at runtime
            client.changeBehavior(to: SomeOtherBehavior())
            client.performBehavior()  // "I have some other behavior"
